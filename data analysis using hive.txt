Step 1: Create Table

CREATE TABLE CovidData (
  location STRING,
  continent STRING,
  total_cases INT,
  total_deaths INT,
  people_vaccinated INT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
STORED AS TEXTFILE;
